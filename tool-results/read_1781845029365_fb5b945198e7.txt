    58→function seedOrders(): Order[] {

   339→export const useLms = create<LmsState>()(
   340→  persist(
   341→    (set, get) => ({
   342→      // --- Nav ---
   343→      view: "home",
   344→      courseSlug: null,
   345→      lessonId: null,
   346→      blogSlug: null,
   347→      certificateId: null,
   348→      adminTab: "overview",
   349→      checkoutCourseId: null,
   350→      searchQuery: "",
   351→      mobileMenuOpen: false,
   352→
   353→      // --- Auth ---
   354→      user: null,
   355→      authOpen: false,
   356→      authMode: "login",
   357→
   358→      // --- Cart ---
   359→      items: [],
   360→      appliedCouponCode: null,
   361→
   362→      // --- Data (seeded) ---
   363→      orders: seedOrders(),
   364→      enrollments: seedEnrollments(),
   365→      notes: [],
   366→      bookmarks: [],
   367→      certificates: seedCertificates(),
   368→      notifications: seedNotifications(),
   369→      announcements: seedAnnouncements(),
   370→      activities: seedActivities(),
   371→      couponList: seedCoupons,
   372→
   373→      // ---------------- Navigation ----------------
   374→      navigate: (view, params) => {
   375→        set({ view, mobileMenuOpen: false, ...params });
   376→        get().scrollTop();
   377→      },
   378→      openCourse: (slug) => {
   379→        set({ view: "course", courseSlug: slug, mobileMenuOpen: false });
   380→        get().scrollTop();
   381→      },
   382→      openLesson: (courseSlug, lessonId) => {
   383→        set({ view: "learn", courseSlug, lessonId, mobileMenuOpen: false });
   384→        get().scrollTop();
   385→      },
   386→      openBlog: (slug) => {
   387→        set({ view: "blog-post", blogSlug: slug, mobileMenuOpen: false });
   388→        get().scrollTop();
   389→      },
   390→      openCertificate: (verifyId) => {
   391→        set({ view: "certificate", certificateId: verifyId, mobileMenuOpen: false });
   392→        get().scrollTop();
   393→      },
   394→      openCheckout: (courseId) => {
   395→        set({ view: "checkout", checkoutCourseId: courseId, appliedCouponCode: null, mobileMenuOpen: false });
   396→        get().scrollTop();
   397→      },
   398→      setAdminTab: (tab) => {
   399→        set({ adminTab: tab });
   400→        get().scrollTop();
   401→      },
   402→      setSearchQuery: (q) => set({ searchQuery: q }),
   403→      setMobileMenuOpen: (open) => set({ mobileMenuOpen: open }),
   404→      scrollTop: () => {
   405→        if (typeof window !== "undefined") window.scrollTo({ top: 0, behavior: "smooth" });
   406→      },
   407→
   408→      // ---------------- Auth ----------------
   409→      setAuthOpen: (open, mode = "login") => set({ authOpen: open, authMode: mode }),
   410→      login: (email, _password) => {
   411→        if (!email || !email.includes("@")) return { ok: false, message: "Enter a valid email." };
   412→        const isAdmin = email.toLowerCase() === "admin@learniverse.io";
   413→        const user: User = isAdmin
   414→          ? DEMO_ADMIN
   415→          : { ...DEMO_USER, email, name: email.split("@")[0].replace(/[^a-zA-Z]/g, " ").trim() || "Student" };
   416→        set({ user, authOpen: false });
   417→        get().addActivity("LOGIN", `${user.name} signed in`);
   418→        return { ok: true, message: "Logged in" };
   419→      },
   420→      signup: (name, email, _password) => {
   421→        if (!name || !email.includes("@")) return { ok: false, message: "Please fill all fields correctly." };
   422→        const user: User = {
   423→          id: `user-${Date.now()}`,
   424→          email,
   425→          name,
   426→          avatar: `https://i.pravatar.cc/120?u=${encodeURIComponent(email)}`,
   427→          role: "STUDENT",
   428→          provider: "email",
   429→          emailVerified: true,
   430→        };
   431→        set({ user, authOpen: false });
   432→        get().addNotification({
   433→          type: "SUCCESS",
   434→          title: "Welcome to Learniverse 🎉",
   435→          body: `Hi ${name}, your account is ready. Start exploring courses!`,
   436→        });
   437→        get().addActivity("SIGNUP", `${name} created an account`);
   438→        return { ok: true, message: "Account created" };
   439→      },
   440→      loginWithGoogle: () => {
   441→        const user: User = { ...DEMO_USER, provider: "google" };
   442→        set({ user, authOpen: false });
   443→        get().addActivity("LOGIN", `${user.name} signed in with Google`);
   444→      },
   445→      logout: () => {
   446→        get().addActivity("LOGOUT", "User signed out");
   447→        set({ user: null, view: "home" });
   448→        get().scrollTop();
   449→      },
   450→      enterAdminDemo: () => {
   451→        set({ user: DEMO_ADMIN, view: "admin", adminTab: "overview", authOpen: false });
   452→        get().addActivity("LOGIN", "Admin signed in (demo)");
   453→        get().scrollTop();
   454→      },
   455→      enterStudentDemo: () => {
   456→        set({ user: DEMO_USER, view: "my-learning", authOpen: false });
   457→        get().addActivity("LOGIN", "Demo student signed in");
   458→        get().scrollTop();
   459→      },
   460→      resetPassword: (email) => {
   461→        if (!email.includes("@")) return { ok: false, message: "Enter a valid email." };
   462→        set({ authOpen: false });
   463→        return { ok: true, message: "Reset link sent (demo)." };
   464→      },
   465→
   466→      // ---------------- Cart ----------------
   467→      addToCart: (courseId) =>
   468→        set((s) =>
   469→          s.items.some((i) => i.courseId === courseId)
   470→            ? s
   471→            : { items: [...s.items, { courseId, addedAt: new Date().toISOString() }] }
   472→        ),
   473→      removeFromCart: (courseId) =>
   474→        set((s) => ({ items: s.items.filter((i) => i.courseId !== courseId) })),
   475→      clearCart: () => set({ items: [], appliedCouponCode: null }),
   476→      isInCart: (courseId) => get().items.some((i) => i.courseId === courseId),
   477→
   478→      // ---------------- Coupons ----------------
   479→      validateCoupon: (code, amount, courseId) => {
   480→        const c = couponMap[code.toUpperCase()] || get().couponList.find((x) => x.code === code.toUpperCase());
   481→        if (!c) return { ok: false, message: "Invalid coupon code.", discount: 0 };
   482→        if (!c.active) return { ok: false, message: "This coupon is no longer active.", discount: 0 };
   483→        const now = Date.now();
   484→        if (new Date(c.startsAt).getTime() > now)
   485→          return { ok: false, message: "This coupon is not yet active.", discount: 0 };
   486→        if (c.expiresAt && new Date(c.expiresAt).getTime() < now)
   487→          return { ok: false, message: "This coupon has expired.", discount: 0 };
   488→        if (c.usageLimit && c.usedCount >= c.usageLimit)
   489→          return { ok: false, message: "This coupon has reached its usage limit.", discount: 0 };
   490→        if (amount < c.minAmount)
   491→          return {
   492→            ok: false,
   493→            message: `Minimum order amount of ₹${c.minAmount.toLocaleString("en-IN")} required.`,
   494→            discount: 0,
   495→          };
   496→        if (c.courseId && courseId && c.courseId !== courseId)
   497→          return { ok: false, message: "This coupon is not valid for this course.", discount: 0 };
   498→        const discount = c.type === "PERCENT" ? Math.round((amount * c.value) / 100) : Math.min(c.value, amount);
   499→        if (discount <= 0) return { ok: false, message: "Coupon discount is zero.", discount: 0 };
   500→        return { ok: true, message: `Coupon applied! You saved ₹${discount.toLocaleString("en-IN")}.`, discount, couponCode: c.code };
   501→      },
   502→      applyCoupon: (code, amount, courseId) => {
   503→        const res = get().validateCoupon(code, amount, courseId);
   504→        if (res.ok) set({ appliedCouponCode: res.couponCode! });
   505→        return res;
   506→      },
   507→      removeCoupon: () => set({ appliedCouponCode: null }),
   508→      toggleCouponActive: (id) =>
   509→        set((s) => ({
   510→          couponList: s.couponList.map((c) => (c.id === id ? { ...c, active: !c.active } : c)),
   511→        })),
   512→      addCoupon: (c) =>
   513→        set((s) => ({
   514→          couponList: [...s.couponList, { ...c, id: `cp-${Date.now()}`, usedCount: 0 }],
   515→        })),
   516→      deleteCoupon: (id) =>
   517→        set((s) => ({ couponList: s.couponList.filter((c) => c.id !== id) })),
   518→
   519→      // ---------------- Orders ----------------
   520→      checkout: (courseId, paymentRef, paymentMethod) => {
   521→        const course = courseMap[courseId];
   522→        const user = get().user || DEMO_USER;
   523→        const appliedCode = get().appliedCouponCode;
   524→        const res = appliedCode ? get().validateCoupon(appliedCode, course.price, courseId) : null;
   525→        const discount = res?.ok ? res.discount : 0;
   526→        const finalAmount = course.price - discount;
   527→        const created = new Date().toISOString();
   528→        const order: Order = {
   529→          id: `ord-${Date.now()}`,
   530→          orderNumber: generateOrderNumber(),
   531→          userId: user.id,
   532→          userEmail: user.email,
   533→          userName: user.name,
   534→          courseId,
